  # Crontab will NOT add new entry, only overwrite the main configuration file.
  
  # Run script.sh every 5 minutes.
  echo "*/5 * * * * script.sh" | crontab
  
  # Run script.sh hourly.
  echo "0 * * * * script.sh" | crontab
  echo "@hourly script.sh" | crontab
  
  # List crontab job.
  crontab -l
