  function get_executing_path()
  {
    local executing_path=$(dirname "$0")              # relative
    executing_path=$( cd "${executing_path}" && pwd ) # absolutized and normalized
    if [ -z "${executing_path}" ]; then
      return 1  # fail
    fi
  }
  export -f get_executing_path
