# Remove duplicates spaces.
	echo "Hello   World" | tr -s ' '

# Remove backspace(BS).
	cat file.txt | tr -d '\b'
	  
# Remove all newlines.
	cat file.txt | tr -d '\n'

# To lowercase.
	echo LOWERCASE | tr '[A-Z]' '[a-z]'
	echo LOWERCASE | tr '[:upper:]' '[:lower:]'
	for f in * ; do mv -v $f `echo $f | tr '[A-Z]' '[a-z]'`; done
